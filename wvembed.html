page HTML "EMBED"

IMPORTANT 1 : le fichier "WvEmbed.js" doit toujours accompagner cette page-HTML-embed
IMPORTANT 2 : paramétrez "WvEmbed.js" - rechercher PARAMETER_TODO dans le script.

Il vous suffit d'ajouter les fonctions qui recevront les requètes de la page HTML principale.

- bleu : à recopier tel quel
- vert : valeur inscrite en paramètres, dans l'appel de la page HTML principale
- violet : valeur récupéré dans le script de la page principal
______________________________________________________________
Exemple de fonction synchrone:

WvEmbedEvent.fonc1 = function(messageP){
        console.log("Oui je suis l'embed 2 "+messageP);    // Placez ici votre script
        return "Voilà le résultat";
};
______________________________________________________________
Exemple de fonction asynchrone: 

WvEmbedEvent.fonc1 = function(messageP,returned){
        setTimeout(
                 function  (messageP,returned){
                        console.log("Oui je suis l'embed 1"+messageP);
                        returned("Voilà le résultat");     // Notez que le retour sera envoyé par : returned(reponse) 
                }                                                            //  fonction callback
       ,         3000, messageP, returned
         );  
};
